543  void process_arg_vector(Vector<std::string> &arg_vector, std::ostream &os)
544  {
545  Vector<std::pair<std::string, Vector<std::string>>> name_svals;
546  Vector<std::string> svals;
547  auto aiter = arg_vector.begin();
548  while (aiter != arg_vector.end() && !is_option(*aiter))
549  ++aiter;
550  while (aiter != arg_vector.end()) {
551  std::string name(aiter->begin() + 2, aiter->end());
552  ++aiter;
553  svals.clear();
554  while (aiter != arg_vector.end() && !is_option(*aiter)) {
555  svals.push_back(*aiter);
556  ++aiter;
557  }
558  name_svals.push_back(std::make_pair(name, svals));
559  }
560 
561  const std::string sval_true("1");
562  for (auto &nsv : name_svals) {
563  auto iter = option_find(nsv.first);
564  if (iter == option_vec_.end()) {
565  internal::program_option_warning(
566  nsv.first, "Unknown option", silent_, os);
567  continue;
568  }
569 
570  bool proc = false;
571  if (nsv.second.size() == 0 && iter->ptr->is_bool()) {
572  proc = process_option(iter, sval_true, os);
573  } else if (nsv.second.size() == 0) {
574  internal::program_option_warning(
575  nsv.first, "No value found", silent_, os);
576  } else if (iter->ptr->is_vector()) {
577  for (const auto &sval : nsv.second)
578  proc = process_option(iter, sval, os);
579  } else {
580  proc = process_option(iter, nsv.second.back(), os) || proc;
581  }
582  if (proc)
583  ++iter->count;
584  }
585 
586  for (auto &option : option_vec_)
587  if (option.count == 0)
588  if (option.ptr->set_default())
589  option.count = 1;
590 
591  if (help())
592  print_help(os);
593  }
