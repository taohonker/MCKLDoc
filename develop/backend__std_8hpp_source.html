43 template <typename IntType>
44 inline void backend_std_range(
45  IntType N, mckl::Vector<IntType> &first, mckl::Vector<IntType> &last)
46 {
47  first.clear();
48  last.clear();
49  const IntType np = std::max(const_one<IntType>(),
50  static_cast<IntType>(std::thread::hardware_concurrency()));
51  if (np == 1) {
52  first.push_back(0);
53  last.push_back(N);
54  return;
55  }
56 
57  const IntType m = N / np;
58  const IntType r = N % np;
59  IntType b = 0;
60  for (IntType id = 0; id != np; ++id) {
61  const IntType n = m + (id < r ? 1 : 0);
62  first.push_back(b);
63  last.push_back(b + n);
64  b += n;
65  }
66 }
