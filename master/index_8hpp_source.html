205  template <typename RandomIter>
206  RandomIter read_index_matrix(MatrixLayout layout, RandomIter first,
207  std::size_t iter_back, std::size_t iter) const
208  {
209  runtime_assert(iter <= iter_back && iter_back < iter_size(),
210  "**ResampleIndex::read_index_matrix** iteration numbers out of "
211  "range");
212 
213  using difference_type =
214  typename std::iterator_traits<RandomIter>::difference_type;
215  const std::size_t N = index_matrix_nrow(iter_back);
216  const std::size_t R = index_matrix_ncol(iter_back, iter);
217 
218  if (layout == RowMajor) {
219  RandomIter back = first + static_cast<difference_type>(R - 1);
220  const index_type *idx = index_[iter_back].data();
221  for (std::size_t i = 0; i != N; ++i)
222  back[static_cast<difference_type>(i * R)] = idx[i];
223  for (std::size_t r = 1; r != R; ++r) {
224  const std::size_t j = iter_back - r;
225  RandomIter last = back;
226  --back;
227  idx = index_[j].data();
228  for (std::size_t i = 0; i != N; ++i) {
229  back[static_cast<difference_type>(i * R)] =
230  idx[static_cast<std::size_t>(
231  last[static_cast<difference_type>(i * R)])];
232  }
233  }
234  }
235 
236  if (layout == ColMajor) {
237  RandomIter back =
238  first + static_cast<difference_type>(N * (R - 1));
239  const index_type *idx = index_[iter_back].data();
240  std::copy_n(idx, N, back);
241  for (std::size_t r = 1; r != R; ++r) {
242  const std::size_t j = iter_back - r;
243  RandomIter last = back;

250  }
