647 template <typename CharT, typename Traits>
648 inline std::basic_ostream<CharT, Traits> &operator<<(
649  std::basic_ostream<CharT, Traits> &os, const MKLStream &stream)
650 {
651  if (!os)
652  return os;
653 
654  std::size_t n = static_cast<std::size_t>(stream.get_size());
655  std::size_t m = sizeof(std::uintmax_t);
656  if (n % m != 0)
657  n += m - n % m;
658  n /= m;
659  Vector<std::uintmax_t> s(n);
660  stream.save_m(reinterpret_cast<char *>(s.data()));
661 
662  os << stream.get_brng() << ' ';
663  os << s;
664 
665  return os;
666 }

873  void operator()(std::size_t n, result_type *r)
874  {
875  internal::size_check<MKL_INT>(n, "MKLEngine::operator()");
876 
877  const std::size_t remain = M_ - index_;
878 
879  if (n < remain) {
880  std::copy_n(buffer_.data() + index_, n, r);
881  index_ += n;
882  return;
883  }

885  std::copy_n(buffer_.data() + index_, remain, r);
886  r += remain;
887  n -= remain;
888  index_ = M_;

890  const std::size_t m = n / M_ * M_;
891  const std::size_t l = n % M_;
892  internal::MKLUniformBits<Bits>::eval(
893  stream_, static_cast<MKL_INT>(m), r);
894  r += m;
895 
896  generate();
897  std::copy_n(buffer_.data(), l, r);
898  index_ = l;
899  }

910  void discard(long long nskip)
911  {
912 
913  if (nskip <= 0)
914  return;
915 
916  const long long remain = static_cast<long long>(M_ - index_);
917  if (nskip <= remain) {
918  index_ += static_cast<std::size_t>(nskip);
919  return;
920  }
921  nskip -= remain;
922  index_ = M_;
923 
924  ::VSLBRngProperties properties;
925  MKLStream::get_brng_properties(stream_.get_brng(), &properties);
926  const int bits = properties.NBits;
927  const long long M = static_cast<long long>(M_);
928  long long m = nskip / M * M;
929  if (Bits >= bits)
930  m *= Bits / bits + (Bits % bits == 0 ? 0 : 1);
931  switch (stream_.get_brng()) {
932  case VSL_BRNG_MCG31: stream_.skip_ahead(m); break;
933  case VSL_BRNG_MRG32K3A: stream_.skip_ahead(m); break;
934  case VSL_BRNG_MCG59: stream_.skip_ahead(m); break;
935  case VSL_BRNG_WH: stream_.skip_ahead(m); break;
936  case VSL_BRNG_MT19937: stream_.skip_ahead(m); break;
937  case VSL_BRNG_SFMT19937: stream_.skip_ahead(m); break;
938  case VSL_BRNG_SOBOL: stream_.skip_ahead(m); break;
939  case VSL_BRNG_NIEDERR: stream_.skip_ahead(m); break;
940 #if INTEL_MKL_VERSION >= 110300
941  case VSL_BRNG_PHILOX4X32X10: stream_.skip_ahead(m); break;
942  case VSL_BRNG_ARS5: stream_.skip_ahead(m); break;
943 #endif
944  default:
945  while (nskip >= M) {
946  generate();
947  nskip -= M;
948  }
949  };
950  generate();
951  index_ = static_cast<std::size_t>(nskip % M);
952  }
