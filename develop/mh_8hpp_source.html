95  template <typename RNGType, typename LogTarget, typename Proposal>
96  std::size_t operator()(RNGType &rng, result_type *x, double *ltx,
97  LogTarget &&log_target, Proposal &&proposal)
98  {
99  U01Distribution<double> u01;
100  double s = ltx == nullptr ? static_cast<double>(log_target(x)) : *ltx;
101  double q = static_cast<double>(proposal(rng, x, y_.data()));
102  double t = static_cast<double>(log_target(y_.data()));
103  double p = t - s + q;
104  double u = std::log(u01(rng));
105 
106  if (u < p) {
107  std::copy(y_.begin(), y_.end(), x);
108  if (ltx != nullptr)
109  *ltx = t;
110  return 1;
111  }
112  return 0;
113  }

116  template <typename RNGType, typename LogTarget, typename Proposal>
117  std::size_t operator()(std::size_t n, RNGType &rng, result_type *x,
118  double *ltx, LogTarget &&log_target, Proposal &&proposal)
119  {
120  if (n == 0)
121  return 0;
122 
123  if (n == 1) {
124  return operator()(rng, x, ltx, std::forward<LogTarget>(log_target),
125  std::forward<Proposal>(proposal));
126  }
127 
128  U01Distribution<double> u01;
129  std::copy_n(x, dim(), x_.data());
130  std::size_t accept = 0;
131  double s = ltx == nullptr ? static_cast<double>(log_target(x)) : *ltx;
132  for (std::size_t i = 0; i != n; ++i) {
133  double q =
134  static_cast<double>(proposal(rng, x_.data(), y_.data()));
135  double t = static_cast<double>(log_target(y_.data()));
136  double p = t - s + q;
137  double u = std::log(u01(rng));
138  if (u < p) {
139  ++accept;
140  s = t;
141  x_.swap(y_);
142  }
143  }
144  if (accept != 0) {
145  std::copy(x_.begin(), x_.end(), x);
146  if (ltx != nullptr)
147  *ltx = s;
148  }
149 
150  return accept;
151  }

155  internal::StaticVector<ResultType, Dim> y_;

598  template <typename RNGType>
599  result_type operator()(RNGType &rng, const result_type *x, result_type *y)
600  {
601  const std::size_t d = dim();
602  normal_mv_(rng, z_.data());
603  exp(d - 1, z_.data(), y);
604  mul(d - 1, x, y, y);
605  mul(d - 1, 1 / x[d - 1], y, y);
606  y[d - 1] = 1;
607  mul(d, 1 / std::accumulate(y, y + d, 0.0), y, y);
608 
609  return q(d, y) - q(d, x);
610  }
611 
612  private:
613  NormalMVDistribution<RealType, Dim> normal_mv_;
614  internal::StaticVector<RealType, Dim> z_;
615 
616  result_type q(std::size_t d, const result_type *x)
617  {
618  result_type slw = 1;
619  result_type sllw = 0;
620  const result_type w = x[d - 1];
621  for (std::size_t i = 0; i != d - 1; ++i) {
622  result_type v = x[i] / w;
623  slw += v;
624  sllw += std::log(v);
625  }
626  sllw -= d * std::log(slw);
627 
628  return sllw;
