340  std::map<std::string, Vector<double>> summary() const
341  {
342  std::map<std::string, Vector<double>> df;
343  Vector<double> data(iter_size());
344 
345  std::copy(size_history_.begin(), size_history_.end(), data.begin());
346  df["Size"] = data;
347 
348  std::copy(resampled_history_.begin(), resampled_history_.end(),
349  data.begin());
350  df["Resampled"] = data;
351 
352  df["ESS"] = ess_history_;
353 
354  for (const auto &m : monitor_) {
355  if (m.second.iter_size() > 0) {
356  for (std::size_t d = 0; d != m.second.dim(); ++d) {
357  std::size_t miter = 0;
358  for (std::size_t i = 0; i != iter_size(); ++i) {
359  if (miter != m.second.iter_size() &&
360  i == m.second.index(miter)) {
361  data[i] = m.second.record(d, miter++);
362  } else {
363  data[i] = const_nan<double>();
364  }
365  }
366  if (m.second.name(d).empty())
367  df[m.first + "." + std::to_string(d)] = data;
368  else
369  df[m.second.name(d)] = data;
370  }
371  }
372  }
373 
374  return df;
375  }

471  void do_resample(double threshold)
472  {
473  size_history_.push_back(size());
474  ess_history_.push_back(particle_.weight().ess());
475 
476  if (ess_history_.back() < size() * threshold) {
477  resampled_history_.push_back(true);
